CREATE DATABASE depanaz_db; USE depanaz_db; CREATE TABLE vehicle_registrations ( id INT AUTO_INCREMENT PRIMARY KEY, first_name VARCHAR(100) NOT NULL, last_name VARCHAR(100) NOT NULL, vehicle_make VARCHAR(100) NOT NULL, vehicle_model VARCHAR(100) NOT NULL, vehicle_color VARCHAR(50) NOT NULL, registration_number VARCHAR(50) NOT NULL UNIQUE, password_hash VARCHAR(255) NOT NULL, registered_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP );
